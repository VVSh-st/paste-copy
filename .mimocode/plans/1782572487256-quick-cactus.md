# Авто-заголовок подвкладок `<12345>`

## Цель
Добавить кнопку авто-заголовка ПОСЛЕ `<12345>`, которая запрашивает у LLM 4 варианта названия текущей подвкладки за один запрос. Пользователь видит первый вариант, нажимая "↺" циклически переключает 2→3→4. При выборе остальные отбрасываются. Кнопка "Авто-заголовок LLM" блока остаётся как есть.

```js
function _shortSubtabLabel(name) {
  if (!name) return '';
  const first = name.split(/\s+/)[0] || '';
  return first.length <= 6 ? first : first.slice(0, 6);
}
```

## Фаза 5: SubtabAutoTitle — логика с 4 вариантами

**Файл**: `llm-features.js`

### 5a. Добавить модуль `SubtabAutoTitle`
Добавить после `AutoTitle` (~строка 485):
```js
const SubtabAutoTitle = (() => {
  let _variants = [];   // 4 варианта из LLM
  let _current = 0;     // индекс текущего показываемого
  let _blockId = null;

  async function autoTitle(blockId) {
    if (!_guard()) return;
    const t = _State.getActive();
    const block = _State.findBlock(t?.blocks ?? [], blockId);
    if (!block) return;

    const sub = block.subtabs?.[block.activeSubtab ?? 0];
    if (!sub) return;
    const text = (sub.value || '').slice(0, 500).trim();
    if (!text) { window.Toast?.show('Вкладка пустая', 'error'); return; }

    _removeExistingPopup();

    // Если уже есть варианты — просто покажем следующий
    if (_variants.length > 0 && _blockId === blockId) {
      _current = (_current + 1) % _variants.length;
      _showSuggestionPopup(blockId, sub);
      return;
    }

    let result;
    try {
      _showThinking('◕ Генерирую варианты...');
      result = await _LLMCore.request({
        messages: [
          { role: 'system', content: _LLMCore.getPrompt('subtab_autotitle') + (_LANG_INSTR ?? '') },
          { role: 'user',   content: text },
        ],
        maxTokens: 200,
        stream: false,
        featureTag: 'subtab_autotitle',
      });
    } catch (e) {
      if (e.name !== 'AbortError') window.Toast?.show('Ошибка: ' + e.message, 'error');
      return;
    } finally { _hideThinking(); }

    if (result == null || typeof result !== 'string') {
      window.Toast?.show('Пустой ответ модели', 'error');
      return;
    }

    _variants = result.trim().split('\n')
      .map(l => l.replace(/^\d+[.):\s]+/, '').replace(/[""]/g, '').trim())
      .filter(l => l.length > 0 && l.length <= 60)
      .slice(0, 4);

    if (_variants.length === 0) {
      window.Toast?.show('Не удалось распарсить варианты', 'error');
      return;
    }

    _blockId = blockId;
    _current = 0;
    _showSuggestionPopup(blockId, sub);
  }

  function _showSuggestionPopup(blockId, sub) {
    // ... popup с текущим вариантом, кнопки: Принять, ↺ Следующий, Отмена
    // Принять: sub.name = _variants[_current], очистить _variants
    // ↺: _current = (_current + 1) % _variants.length, перерисовать
    // Отмена: очистить _variants, закрыть
  }

  return { autoTitle };
})();
```

## Фаза 6: Парсинг короткого названия

При `sub.name = _variants[_current]`:
- В `buildTabs()` и `patchSubtab()` уже будет вызываться `_shortSubtabLabel(sub.name)` → первое слово ≤6 символов
- Тултип marquee получает полное `sub.name` — показывает всю фразу
- Никакого дополнительного парсинга не нужно

## Верификация

1. Открыть текстовый блок → подвкладки шире, влезает 7-8 символов
2. Двойной клик по подвкладке → ренейм работает, поле шире
3. Кнопка `⊕` после `<12345>` → клик → LLM-запрос → popup с вариантом
4. Нажать `↺` → следующий вариант (2, 3, 4, затем цикл)
5. Нажать "Принять" → подвкладка показывает первое слово ≤6 символов
6. Навести на подвкладку → тултип с полным названием и marquee
7. Кнопка "Авто-заголовок LLM" блока → работает как раньше (устанавливает `b.title`)
8. Новый промпт виден в настройках LLM → Служебные → "Авто-заголовок вкладки"
